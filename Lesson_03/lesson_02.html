#include < math.h >
using namespace std;
int main() {
float a, b, c;
cout << "PT bac 2 co dang: ax^2 + bx + c =0 ( a khac 0 )" << endl;
do {
cout << "Nhap a: ";
cin >> a;
cout << "Nhap b: ";
cin >> b;
cout << "Nhap c: ";
cin >> c;
} while (a == 0);
float denta = b * b - 4 * a * c;
if (denta > 0) {
cout << "Phuong trinh co 2 nghiem phan biet" << endl;
cout << (-b + sqrt(denta) / (2 * a)) << endl;
cout << (-b - sqrt(denta) / (2 * a)) << endl;
}
else if (denta == 0) {
cout << "PT co 1 nghiem kep: " << -b / 2*a << endl;
}
else
{
cout << "Phuong trinh vo nghiem" << endl;
}
system("pause");
return 0;
}
